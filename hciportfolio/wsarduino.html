Assignment 3C

For assignment 3 we had two blinking LEDs, initially blinking simultaneously, then alternating. Finally, assignment 3c was to make the green LED blink twice as fast as the yellow one. The results are below.

void setup() {
  pinMode(12, OUTPUT);
  pinMode(13, OUTPUT);
}

void loop() {
  digitalWrite(12, HIGH);
  digitalWrite(13, HIGH); 
  delay(500);
  digitalWrite(13, LOW);
  delay(500);
  digitalWrite(12, LOW);
  digitalWrite(13, HIGH);
  delay(500);
  digitalWrite(13, LOW);
  delay(500);
}

Assignment 4C

Assignment 4 introduced analog writing by making the LEDs turn on with half brightness. Assignment 4c asked us to extend the code so that one LED slowly fades in while the other slowly fades out, at the same. The results are below.

int greenLedPin = 10;
int yellowLedPin = 9;
        
void setup() {
  pinMode(greenLedPin, OUTPUT);
  pinMode(yellowLedPin, OUTPUT);
}

void loop() {
  for (int brightness = 0; brightness <256; brightness++) {
   analogWrite(greenLedPin, brightness);
   analogWrite(yellowLedPin, 255-brightness);
   delay(10);
  }
}

Assignment 6C

Assignment 6 was similar to assignment 4, but with the difference of having a potentiometer controlling the fadding of the LEDs. Assignment 6c requested us to change the code so that one LED fades in while the other one fades out, both controlled by the potentiometer. The results are below.

float sensorValue = 0;
int sensorPin = A0;
int greenLedPin= 10;
int yellowLedPin= 9;
        
void setup() {
  pinMode(sensorPin, INPUT);
  Serial.begin(9600);
  pinMode(greenLedPin, OUTPUT);
  pinMode(yellowLedPin, OUTPUT);
}

void loop() {
  sensorValue = analogRead(sensorPin);
  Serial.println(sensorValue);
  sensorValue= (sensorValue/1023)*255;
  analogWrite(greenLedPin, sensorValue);
  analogWrite(yellowLedPin, 255-sensorValue);
}

#include "LedControl.h"
LedControl lc = LedControl(12,11,10,1);

int sensorValue = 0;
int sensorPin = A0;
int tempi = 0;
int leds[64][2];
int count = 0;



void setup() {
  lc.shutdown(0, false);
  lc.setIntensity(0, 8);
  lc.clearDisplay(0);
  pinMode(sensorPin, INPUT);
  
  Serial.begin(9600);

  for (int a = 0; a <= 7; a++) {
    for (int b = 0; b <= 7; b++) {
      leds[count][0] = {a};
      leds[count][1] = {b};
      count++;
    }
  }
}

void loop(){
  sensorValue = analogRead(sensorPin);
  int i = map(sensorValue, 0, 1023, 0, 63);

  //using an if statement to prevent the matrix from updating all the time
  if (tempi != i) {
   lc.clearDisplay(0);
   for (int l = 0; l <= i-1; l++) {
     lc.setLed(0, leds[l][1], leds[l][0], true);
   }   
   tempi =1;
  }
  
}
